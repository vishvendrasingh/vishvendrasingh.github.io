Hello friends here is the php script to unzip .zip files.

<?php
$zip = new ZipArchive;
$res = $zip->open('latest.zip');
if ($res === TRUE) {
$zip->extractTo('./');
$zip->close();
echo 'woot!';
} else {
echo 'doh!';
}
?>
